Was liefern folgende Varianten der obigen Anfrage?

SELECT C.Name, COALESCE (M.Type, 'not a member') AS UNStatus
FROM Country C LEFT OUTER JOIN IsMember M 
ON (C.Code = M.Country)
WHERE M.Organization = 'UN'
SELECT C.Name, COALESCE (M.Type, 'not a member') AS UNStatus
FROM Country C LEFT OUTER JOIN IsMember M 
ON (C.Code = M.Country)
WHERE M.Organization = 'UN' OR M.Organization IS NULL
