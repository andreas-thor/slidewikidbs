Bevölkerungsdichte für jedes Land 

SELECT Name, Population/Area  AS Dichte
FROM Country

Bevölkerungsdichte kategorisiert (hoch, mittel, niedrig) für jedes Land 

SELECT Name, 
  CASE
    WHEN Population/Area > 500 THEN 'hoch' 
    WHEN Population/Area > 100 THEN 'mittel'
    ELSE 'niedrig'
  END AS DichteKategorie
FROM Country
WHERE Area > 0 
ORDER BY DichteKategorie, Name
