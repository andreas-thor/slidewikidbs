Länder Europas 

SELECT Name, Continent, Percentage
FROM Country, Encompasses
WHERE Country.Code = Encompasses.Country
AND Continent = 'Europe'

Länder, die nur zu einem Teil in Europa liegen

SELECT C.Name, E.Percentage
FROM Country C 
JOIN Encompasses E
ON (C.Code = E.Country)
WHERE E.Continent = 'Europe' 
AND E.Percentage < 100
